gdp_effect = {
	#main groupings
	clear_variable = { industrial_complex_total }
	clear_variable = { Urban_Center_total }
	clear_variable = { infrastructure_total }
	clear_variable = { dockyard_total }
	clear_variable = { synthetic_refinery }
	clear_variable = { rocket_site_total }
	clear_variable = { nuclear_reactor_total }
#manipulate groupings
	clear_variable = { manipulate_ic_total }
	clear_variable = { manipulate_uc_total }
	clear_variable = { manipulate_infra_total }
	clear_variable = { manipulate_dockyard_total }
	clear_variable = { manipulate_synthetic_refinery_total }
	clear_variable = { manipulate_rocket_site_total }
	clear_variable = { manipulate_nuclear_reactor_total }

	every_owned_state = {
		add_to_variable = { ROOT.industrial_complex_total = building_level@industrial_complex }  
		add_to_variable = { ROOT.Urban_Center_total = building_level@Urban_Center }
		add_to_variable = { ROOT.infrastructure_total = building_level@infrastructure }
		add_to_variable = { ROOT.dockyard_total = building_level@dockyard }
		add_to_variable = { ROOT.synthetic_refinery_total = building_level@synthetic_refinery }
		add_to_variable = { ROOT.rocket_site_total = building_level@rocket_site }
		add_to_variable = { ROOT.nuclear_reactor_total = building_level@nuclear_reactor }
	}
	
	set_variable = {
			var = manipulate_ic_total
			value = industrial_complex_total
			
		}
		
	set_variable = {
			var = manipulate_uc_total
			value = Urban_Center_total
			
		}
	
	set_variable = {
			var = manipulate_infra_total
			value = infrastructure_total
			
		}
	
	set_variable = {
			var = manipulate_dockyard_total
			value = dockyard_total
			
		}
		
	set_variable = {
			var = manipulate_synthetic_refinery_total
			value = synthetic_refinery_total
			
		}
	
	set_variable = {
			var = manipulate_rocket_site_total
			value = rocket_site_total
			
		}
		
	set_variable = {
			var = manipulate_nuclear_reactor_total
			value = nuclear_reactor_total
			
		}
}
